def factorial(n):
  if n == 0:
      return 1
  else:
      result = 1
      for i in range(1, n + 1):
          result *= i
      return result

def calcular_combinaciones_personajes(personajes, equipo):
  if personajes < equipo:
      return 0
  else:
      combinaciones = factorial(personajes) // (factorial(equipo) * factorial(personajes - equipo))
      return combinaciones

personajes = 8
equipo = 3
combinaciones = calcular_combinaciones_personajes(personajes, equipo)
print(f"El nÃºmero total de combinaciones posibles de equipos de aventureros es: {combinaciones}")
